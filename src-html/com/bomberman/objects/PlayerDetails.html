001package com.bomberman.objects;
002import com.badlogic.gdx.math.Vector2;
003import com.bomberman.utils.Directions;
004
005/**
006 * @author Maciej Parandyk
007 * @author Mateusz Kloc
008 */
009
010public enum PlayerDetails {
011    //playable character specification
012    PLAYER_ONE(100.0f,"Bman_F_f0",0.09f,8,"Bman_F_f0",0.09f,8,"Bman_B_f0",0.09f,8,new Vector2(40.0f,60.0f),new Vector2(1.0f,1.0f),
013            new Vector2(0.4f,0.5f), AbstractGameObject.Alignment.CENTER),
014
015    //Non-playable character specification
016    NPC(100.0f,"Creep_F_f0",0.09f,6,"Creep_S_f0",0.09f,7,"Creep_B_f0",0.09f,6,new Vector2(35.0f,35.0f),new Vector2(1.0f,1.0f),
017            new Vector2(0.8f,0.8f), AbstractGameObject.Alignment.CENTER);
018
019    private PlayerDetails(float moveSlide,String frontAnimationName,float frontFrameDuration,int frontNbrOfFrames,String sideAnimationName,
020                          float sideFrameDuration, int sideNbrOfFrames, String backAnimationName,float backFrameDuration,
021                          int backNbrOfFrames,Vector2 dimensions,Vector2 scale,
022                          Vector2 reductionBox,AbstractGameObject.Alignment alignment){
023        this.objectMoveSlide=moveSlide;
024        this.objectFrontAnim=frontAnimationName;
025        this.objectSideAnim=sideAnimationName;
026        this.objectBackAnim=backAnimationName;
027        this.objectFrontFrameDuration=frontFrameDuration;
028        this.objectSideFrameDuration=sideFrameDuration;
029        this.objectBackFrameDuration=backFrameDuration;
030        this.objectFrontNumberOfFrames=frontNbrOfFrames;
031        this.objectSideNumberOfFrames=sideNbrOfFrames;
032        this.objectBackNumberOfFrames=backNbrOfFrames;
033        this.objectDimension=dimensions;
034        this.objectScale=scale;
035        this.objectReductionRates=reductionBox;
036        this.objectAlignment=alignment;
037    }
038
039    private final float objectMoveSlide;
040    private final String objectFrontAnim;
041    private final String objectSideAnim;
042    private final String objectBackAnim;
043    private final float objectFrontFrameDuration;
044    private final float objectSideFrameDuration;
045    private final float objectBackFrameDuration;
046    private final int objectFrontNumberOfFrames;
047    private final int objectSideNumberOfFrames;
048    private final int objectBackNumberOfFrames;
049    private final Vector2 objectDimension;
050    private final Vector2 objectScale;
051    private final Vector2 objectReductionRates;
052    private AbstractGameObject.Alignment objectAlignment;
053
054    public float slideDistance() {
055        return objectMoveSlide;
056    }
057
058    public String animation(Directions animationType) {
059        String tmp;
060        switch(animationType){
061            case DOWN:
062                tmp=objectFrontAnim;
063                break;
064            case LEFT:
065            case RIGHT:
066                tmp=objectSideAnim;
067                break;
068            case UP:
069                tmp=objectBackAnim;
070                break;
071                default:
072                    tmp=null;
073        }
074        return tmp;
075    }
076
077    /**
078     * Pobranie długości trwania pojedyńczej klatki animacji w zależnosci od typu animacji obiektu
079     * @param animationType kierunek animacji/skierowanie gracza
080     * @return długość trwania pojedyńczej klatki
081     */
082    public float frameDuration(Directions animationType) {
083        float duration;
084        switch(animationType){
085            case DOWN:
086                duration= this.objectFrontFrameDuration;
087                break;
088            case LEFT:
089            case RIGHT:
090                duration= this.objectSideFrameDuration;
091                break;
092            case UP:
093                duration= this.objectBackFrameDuration;
094                break;
095            default:
096               duration= 1.0f;
097        }
098        return duration;
099    }
100
101    public Vector2 dimensions() {
102        return objectDimension;
103    }
104
105    public Vector2 scale() {
106        return objectScale;
107    }
108
109    public Vector2 reductionRates() {
110        return objectReductionRates;
111    }
112
113    public AbstractGameObject.Alignment alignment() {
114        return objectAlignment;
115    }
116
117    /**
118     * Podaje ilość klatek przypadających na daną animację
119     * @param animationType kierunek animacji
120     * @return ilość klatek przypadających na pełną animację
121     * @return ilość klatek przypadających na pełną animację
122     */
123    public Integer animationFrames(Directions animationType) {
124        Integer tmp;
125        switch(animationType){
126            case DOWN:
127                tmp=this.objectFrontNumberOfFrames;
128                break;
129            case LEFT:
130            case RIGHT:
131                tmp=this.objectSideNumberOfFrames;
132                break;
133            case UP:
134                tmp=this.objectBackNumberOfFrames;
135                break;
136            default:
137                tmp=0;
138        }
139        return tmp;
140    }
141}
